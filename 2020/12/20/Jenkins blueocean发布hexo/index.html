Jenkins blueoceanеҸ‘еёғhexo

Posted on 2020-12-20 Edited on 2021-02-28 In application Disqus:

еҺҹеӣ 

д№ӢеүҚдёҖзӣҙз”Ёtravis-ciеҒҡhexoзҡ„еҸ‘еёғпјҢдҪҶжҳҜд»ҺдёӨдёӘжңҲд№ӢеүҚејҖе§ӢпјҢдёҚзҹҘйҒ“дёәд»Җд№ҲжҸҗдәӨдәҶд»Јз ҒеҗҺпјҢtravis-ciиҷҪз„¶и§ҰеҸ‘дәҶпјҢдҪҶдёҖзӣҙеҚЎpendingзҠ¶жҖҒпјҢдёҖзӣҙдёҚиҝӣиЎҢжһ„е»әгҖӮеҜјиҮҙйңҖиҰҒеңЁжң¬жңәжһ„е»әhexoпјҢз„¶еҗҺжҺЁйҖҒеҲ°masterеҲҶж”ҜпјҢеүҚеҮ зҜҮе…ідәҺconcourseзҡ„е°ұжҳҜжүӢеҠЁжһ„е»әhexoзӣ®еҪ•з»“жһ„иҝӣиЎҢеҸ‘еёғзҡ„гҖӮ

е®һеңЁдёҚиғҪеҝҚдәҶпјҢеүҚеҮ еӨ©йҳҝйҮҢдә‘д№°дәҶеҸ°2c4gзҡ„жңәеҷЁпјҢеҮҶеӨҮиҮӘе·ұеј„дёӘjenkinsпјҢеҒҡжһ„е»әеҸ‘еёғгҖӮ

еӣ дёәд№ӢеүҚе…¬еҸёд№ҹз®ҖеҚ•жҺҘи§ҰиҝҮjenkinsпјҢдёҚиҝҮйӮЈдјҡйғҪжҳҜйЎөйқўзӮ№зӮ№зӮ№пјҢиҝҷж¬ЎеҮҶеӨҮеҶҷpipelineиҝӣиЎҢжһ„е»әгҖӮ

жҲ‘д№ҹжҳҜжүҚзҹҘйҒ“пјҢjenkinsжңүдёӘblue oceanзҡ„з•ҢйқўпјҢдё“й—Ёз”ЁжқҘеј„pipelineгҖӮиҖҢдё”д№ҹжҳҜеҸӘиҰҒе°Ҷpipelineж–Үд»¶(й»ҳи®ӨдёәJenkinsfile,еҸҜд»ҘеңЁжһ„е»әйЎ№зӣ®зҡ„Build ConfigurationйҮҢйқўдҝ®ж”№)ж”ҫзҪ®дәҺд»“еә“ж №зӣ®еҪ•пјҢдҫҝеҸҜд»ҘдҪҝз”ЁJenkinsfileе®ҡд№үзҡ„pipelineиҝӣиЎҢжһ„е»әдәҶгҖӮжҲ‘жңҖж—©йӮЈдјҡжҺҘи§Ұзҡ„gitlabе’Ңtravis-ciд№ҹжҳҜиҝҷд№ҲдёӘжөҒзЁӢгҖӮиҝҷж ·зҡ„иҜқе°ұеҸҜд»ҘеңЁд»Јз Ғд»“еә“зӣҙжҺҘз»ҙжҠӨpipelineдәҶгҖӮ

жӯӨж¬Ўзј–еҶҷзҡ„pipelineдёәDeclarative Pipeline

еҸ‘еёғжөҒзЁӢ

еҸ‘еёғжөҒзЁӢиҝҳжҳҜеҸӮз…§д№ӢеүҚtravis-ciзј–еҶҷзҡ„жөҒзЁӢпјҢеҸӘдёҚиҝҮиҝҷж¬Ўж”№жҲҗз”ЁjenkinsеҺ»е®һзҺ°дәҶгҖӮ

  1. жӢүеҸ–hexoеҲҶж”Ҝе’ҢmasterеҲҶж”Ҝ
  2. еңЁhexoеҲҶж”ҜиҝӣиЎҢhexoзӣ®еҪ•жһ„е»ә
  3. е°ҶmasterеҲҶж”Ҝзҡ„.gitзӣ®еҪ•ж•ҙдёӘеӨҚеҲ¶еҲ°hexoеҲҶж”ҜдёӢзҡ„publicзӣ®еҪ•дёӯ
  4. е°Ҷpublicдёӯзҡ„ж–Үд»¶ејәжҺЁиҮіmasterеҲҶж”ҜпјҢе®ҢжҲҗеҸ‘еёғ

е…¶д»–еҠҹиғҪ

  1. и§ҰеҸ‘ејҸжһ„е»ә
  2. йӮ®д»¶йҖҡзҹҘ

е…·дҪ“ж“ҚдҪң

еүҚжңҹжӯҘйӘӨ

  1. еҮҶеӨҮжҸ’д»¶пјҢжҖ»д№ӢиҰҒз”Ёд»Җд№Ҳе°ұиЈ…д»Җд№ҲжҸ’д»¶пјҢblueoceanжҳҜеҝ…еӨҮзҡ„пјҢExtended E-mail Notificationд№ҹеҫ—еҮҶеӨҮпјҢpipelineдјҡз”ЁExtended E-mail Notificationж–№ејҸеҸ‘йҖҒгҖӮ
  2. еңЁgithubеҮҶеӨҮPersonal access tokensжіЁж„ҸжқғйҷҗпјҢз”ЁдәҺblueoceanеҜ№д»“еә“зҡ„ж“ҚдҪңгҖӮдё”еңЁjenkinsи®ҫзҪ®йҮҢйқўй…ҚзҪ®GithubжңҚеҠЎеҷЁгҖӮ(зҪ‘дёҠж•ҷзЁӢдёҖеӨ§жҠҠ)
  3. еҮҶеӨҮдёҖдёӘйӮ®з®ұпјҢе№¶и®ҫзҪ®е…¶иғҪеӨҹдҪҝз”Ёи„ҡжң¬зҡ„ж–№ејҸеҸ‘йӮ®д»¶пјҢжҜ”еҰӮqqзҡ„жҳҜејҖеҗҜIMAPзӢ¬з«ӢtokenгҖӮ(дёҖејҖе§ӢжҲ‘з”Ёзҡ„йҳҝйҮҢйӮ®з®ұпјҢе®һеңЁжІЎжүҫеҲ°еңЁе“ӘйҮҢпјҢеҗ¬иҜҙз”Ё163зҡ„д№ҹеҸҜд»ҘгҖӮ)з”ұдәҺECSжҳҜйҳҝйҮҢдә‘зҡ„пјҢжүҖд»Ҙ25з«ҜеҸЈж— жі•дҪҝз”ЁпјҢqqзҡ„иҜқеҫ—иө°587 smtpз«ҜеҸЈгҖӮ
  4. ж №жҚ®йӮ®з®ұдҝЎжҒҜй…ҚзҪ®еҘҪExtended E-mail NotificationйҮҢйқўзҡ„еҶ…е®№пјҢSMTP server,SMTP Portдё»иҰҒжҳҜиҝҷдёӨйғЁеҲҶгҖӮ
  5. jenkinsи®ҫзҪ®йҮҢйқўзҡ„Jenkins LocationпјҢй…ҚзҪ®еҘҪеҸ‘йҖҒйӮ®з®ұзҡ„ең°еқҖгҖӮ
  6. еңЁgithubдёҠйңҖиҰҒжһ„е»әзҡ„repoд»“еә“дёӯй…ҚзҪ®еҘҪwebhookпјҢз”ЁдәҺжҸҗдәӨеҗҺеҮәеҸ‘еӣһи°ғз»ҷjenkinsпјҢjenkins webhookең°еқҖдёҖиҲ¬дёәвҖқjenkinsжңҚеҠЎеҷЁең°еқҖ/github-webhookвҖқгҖӮ
  7. ж №жҚ®blueoceanзҡ„еј•еҜјпјҢйҖүжӢ©иҰҒжһ„е»әзҡ„githubйЎ№зӣ®гҖӮ

зј–еҶҷJenkinsfile

  1. е°ҶйңҖиҰҒжһ„е»әзҡ„branchиҝҪеҠ иҜҘJenkinsfile
  2. еҶ…е®№еҰӮдёӢ
pipeline {
    agent {
        label 'master'
    }

    stages {
        // е№¶иЎҢжӢүеҸ–hexoе’ҢmasterеҲҶж”Ҝ
        stage('Parellel get repo') {
            parallel {
                stage('Fetch hexo barnch') {
                    environment {
                        NVM_DIR = "/var/lib/jenkins/.nvm"
                    }
                    steps {
                        // жҢҮе®ҡе·ҘдҪңзӣ®еҪ•дёәhexo-branch, е№¶е°ҶhexoеҲҶж”ҜжӢүеҸ–еҲ°hexo-branchзӣ®еҪ•пјҢеҰӮжһңжІЎжңүиҝҷдёӘзӣ®еҪ•еҲҷеҲӣе»әгҖӮ
                        dir('hexo-branch') {
                            // credentialsIdеңЁjenkinsй…ҚзҪ®зҡ„credential managerдёӯеҸҜд»ҘжүҫеҲ°пјҢд№ҹе°ұжҳҜд№ӢеүҚй…ҚзҪ®зҡ„Personal access tokensеҜ№еә”зҡ„id
                            git branch: 'hexo', credentialsId: '0cc091e1-b69f-4e1d-b8c6-b7a9df25e438', url: 'https://github.com/huangyisan/myblog.git'
                        }
                    }

                }
                stage('Fetch master branch') {
                    steps{
                        dir('master-branch') {
                            git branch: 'master', credentialsId: '0cc091e1-b69f-4e1d-b8c6-b7a9df25e438', url: 'https://github.com/huangyisan/myblog.git'
                        }
                    }
                }
            }
        }

        stage('Build hexo') {
            steps {
                dir('hexo-branch') {
                    // еӣ дёәжҲ‘з”Ёзҡ„жҳҜnvmз®ЎзҗҶзҡ„nodeпјҢжүҖд»Ҙеҫ—жҸҗеүҚеңЁlinuxзҡ„jenkinsз”ЁжҲ·дёӢе®үиЈ…nvmпјҢиҝҷиҫ№жҳҜеј•е…ҘзҺҜеўғпјҢе№¶дё”жһ„е»әhexo
                    sh '[ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh" && npm install && hexo clean && hexo g'
                }
            }
        }

        stage('cp master .git to hexo-branch/public'){
            steps{
                // жҢҮе®ҡе·ҘдҪңзӣ®еҪ•дёәmaster-branchпјҢе№¶дё”е°Ҷ.gitж–Үд»¶cpеҲ°hexo-branchдёӢзҡ„publicдёӯ
                dir('master-branch') {
                    sh 'cp -a ./.git ../hexo-branch/public'
                }
            }
        }

        stage('set git user info') {
            steps {
                dir('hexo-branch') {
                    // и®ҫе®ҡжҸҗдәӨд»Јз Ғдәәзҡ„дҝЎжҒҜ
                    sh 'cd ./public && git config user.name  "huangyisan" && git config user.email "anonymousyisan@gmail.com"' 
                }
            }
        }

        stage ('git add && commit') {
            steps {
                dir('hexo-branch') {
                    // Git commitдҝЎжҒҜ
                    sh 'cd ./public && git add . && git commit -m "Jenkins CI Auto Builder at `date +"%Y-%m-%d %H:%M"`" '
                }
            }
        }

        stage ('Push public to Master branch') {
            environment {
                // е®ҡд№үдәҶдёҖдёӘеҸҳйҮҸGH_REF
                GH_REF="github.com/huangyisan/myblog.git"
            }
            steps {
                dir('hexo-branch') {
                    // д»Һcredential managerдёӯиҺ·еҸ–idдёә'e5d2d117-b5ab-4dc5-9e07-a5e96bfb6e31'зҡ„еҜҶй’ҘпјҢиөӢеҖјз»ҷTOKEN
                    withCredentials([string(credentialsId: 'e5d2d117-b5ab-4dc5-9e07-a5e96bfb6e31', variable: 'TOKEN')]) {
                        // иҝӣе…Ҙpublicзӣ®еҪ•еҗҺпјҢејәжҺЁmasterеҲҶж”Ҝ
                        sh 'cd public && git push --force --quiet "https://${TOKEN}@${GH_REF}" master:master'
                    }
                }
            }
        }
    }
    // postдёәжһ„е»әз»“жқҹеҗҺзҡ„ж“ҚдҪң
    post {
        // alwaysиЎЁзӨәж— и®әжһ„е»әз»“жһ„жҳҜжҲҗеҠҹиҝҳжҳҜеӨұиҙҘпјҢдәҰжҲ–е…¶д»–йғҪжү§иЎҢ
        always {
            dir('hexo-branch') {
                script {
                    // иҺ·еҸ–commit idеҸ·, и®ҫе®ҡйӮ®д»¶зҡ„зұ»еһӢпјҢеҸ‘йҖҒж–№пјҢдё»йўҳпјҢеҶ…е®№зӯү
                    def COMMIT_ID = ""
                    COMMIT_ID = sh(returnStdout: true, script:'git log --pretty=format:"%h" -1')

                    def mimeType = 'text/html'

                    def to = 'anonymousyisan@gmail.com'

                    def subject = 'гҖҗжһ„е»әйҖҡзҹҘгҖ‘$PROJECT_NAME - ' + "${COMMIT_ID}" +  ' - Build # $BUILD_NUMBER - $BUILD_STATUS!'

                    def body = 
                    '''
<html>

з•Ҙ....

</html>
                    '''
                // жү§иЎҢеҸ‘йҖҒйӮ®д»¶еҮҪж•°
                emailext(
                    to: to,
                    subject: subject,
                    mimeType: mimeType,
                    body: body
                )
                }
            }
        }
    }
}
```python
